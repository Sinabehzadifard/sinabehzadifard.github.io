mobileMenu.classList.add('opacity-0', 'invisible'); // Update the menu icon mobileMenuButton.innerHTML = ''; } else { mobileMenu.classList.remove('opacity-0', 'invisible'); mobileMenu.classList.add('opacity-100', 'visible');
